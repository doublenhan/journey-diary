### 4. Firestore Security Rules

```javascript
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // Helper functions
    function isAuthenticated() {
      return request.auth != null;
    }
    
    function isOwner(userId) {
      return isAuthenticated() && request.auth.uid == userId;
    }
    
    function getUserData(userId) {
      return get(/databases/$(database)/documents/dev_users/$(userId)).data;
    }
    
    function isAdmin() {
      return isAuthenticated() && 
             getUserData(request.auth.uid).role in ['Admin', 'SysAdmin'];
    }
    
    function isSysAdmin() {
      return isAuthenticated() && 
             getUserData(request.auth.uid).role == 'SysAdmin';
    }
    
    // Users collection (production)
    match /users/{userId} {
      allow read: if isAuthenticated() && (isOwner(userId) || isAdmin());
      allow create: if isAuthenticated();
      allow update: if isOwner(userId) || isSysAdmin();
      allow delete: if isSysAdmin();
    }
    
    // Dev users collection
    match /dev_users/{userId} {
      allow read: if isAuthenticated() && (isOwner(userId) || isAdmin());
      allow create: if isAuthenticated();
      allow update: if isOwner(userId) || isSysAdmin();
      allow delete: if isSysAdmin();
    }
    
    // Memories collection (production)
    match /memories/{memoryId} {
      allow read: if isAuthenticated() && 
                     (resource.data.userId == request.auth.uid || isAdmin());
      allow create: if isAuthenticated() && 
                       request.resource.data.userId == request.auth.uid;
      allow update, delete: if isAuthenticated() && 
                               resource.data.userId == request.auth.uid;
    }
    
    // Dev memories collection
    match /dev_memories/{memoryId} {
      allow read: if isAuthenticated() && 
                     (resource.data.userId == request.auth.uid || isAdmin());
      allow create: if isAuthenticated() && 
                       request.resource.data.userId == request.auth.uid;
      allow update, delete: if isAuthenticated() && 
                               resource.data.userId == request.auth.uid;
    }
    
    // Couples collection (production & dev)
    match /{prefix}couples/{coupleId} {
      allow read: if isAuthenticated() && 
                     (resource.data.user1Id == request.auth.uid || 
                      resource.data.user2Id == request.auth.uid);
      allow create: if isAuthenticated();
      allow update: if isAuthenticated() && 
                       (resource.data.user1Id == request.auth.uid || 
                        resource.data.user2Id == request.auth.uid);
      allow delete: if isAuthenticated() && 
                       (resource.data.user1Id == request.auth.uid || 
                        resource.data.user2Id == request.auth.uid);
    }
    
    // Couple invitations (production & dev)
    match /{prefix}coupleInvitations/{inviteId} {
      allow read: if isAuthenticated() && 
                     (resource.data.senderId == request.auth.uid || 
                      resource.data.receiverId == request.auth.uid);
      allow create: if isAuthenticated() && 
                       request.resource.data.senderId == request.auth.uid;
      allow update: if isAuthenticated() && 
                       (resource.data.senderId == request.auth.uid || 
                        resource.data.receiverId == request.auth.uid);
      allow delete: if isAuthenticated() && 
                       resource.data.senderId == request.auth.uid;
    }
    
    // Shared memories (production & dev)
    match /{prefix}sharedMemories/{sharedId} {
      allow read: if isAuthenticated() && 
                     (resource.data.ownerId == request.auth.uid || 
                      resource.data.sharedWithId == request.auth.uid);
      allow create: if isAuthenticated() && 
                       request.resource.data.ownerId == request.auth.uid;
      allow update: if isAuthenticated() && 
                       resource.data.ownerId == request.auth.uid;
      allow delete: if isAuthenticated() && 
                       resource.data.ownerId == request.auth.uid;
    }
    
    // System stats (admin only)
    match /system_stats/{document} {
      allow read: if isSysAdmin();
      allow write: if false; // Only cloud functions can write
    }
  }
}
```

## ğŸ”§ Troubleshooting Guide

### Common Issues

#### 1. **403 Forbidden on Stats Calculation**
**Cause**: User document not found or wrong role  
**Fix**: Check collection name (dev_users vs users)
```bash
# Verify user exists
firebase firestore:get dev_users/{userId}

# Check role
firebase firestore:get dev_users/{userId} --field role
```

#### 2. **Cloudinary Delete Fails**
**Cause**: Wrong credentials or publicId  
**Fix**: Verify environment variables
```bash
# Check function config
firebase functions:config:get

# Test credentials
curl -X DELETE \
  "https://api.cloudinary.com/v1_1/dhelefhv1/image/destroy" \
  -d "public_id=test&api_key=[YOUR_API_KEY]&api_secret=[YOUR_API_SECRET]"
```

#### 3. **Stats Not Updating**
**Cause**: system_stats/function_calls document missing  
**Fix**: Create memory to trigger tracking
```javascript
// Document will be auto-created on first function call
// Or manually create:
firebase firestore:set system_stats/function_calls '{}'
```
